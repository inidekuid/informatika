# Program klasifikasi jenis segitiga
a = int(input("Masukkan sisi pertama: "))
b = int(input("Masukkan sisi kedua: "))
c = int(input("Masukkan sisi ketiga: "))

# Memeriksa apakah sisi-sisi dapat membentuk segitiga
if a + b > c and a + c > b and b + c > a:
    # Jika bisa, periksa jenisnya
    if a == b == c:
        print("Segitiga Sama Sisi")
    elif a == b or b == c or a == c:
        print("Segitiga Sama Kaki")
    else:
        print("Segitiga Sembarang")
else:
    # Jika tidak, beri tahu pengguna
    print("Bukan Segitiga")
